const APP_VERSION = '2.0.0';
const TZ = Session.getScriptTimeZone() || 'America/Santiago';

const SHEETS = {
  proveedores: 'Proveedores',
  contactos: 'Contactos',
  hoteleria: 'Hoteleria',
  acuerdos: 'Acuerdos',
  programas: 'Programas',
  catalogos: 'Catalogos',
  cambios: 'Cambios_Log'
};

const HEADERS = {
  Proveedores: [
    'proveedor_id','rut_empresa','razon_social','nombre_fantasia','localidad','direccion','correo_empresa','fono_empresa',
    'giros_sii','rubros_norm','actividad_principal','descripcion_general','plataformas_mineras','categoria_sii',
    'estado_facturacion','agrupacion_gremial','servicios_am','estado_registro','fuente','created_at','updated_at','deleted_at','observaciones'
  ],
  Contactos: [
    'contacto_id','proveedor_id','rut_persona','nombre','cargo','correo','fono','principal','estado_registro','created_at','updated_at','deleted_at','observaciones'
  ],
  Hoteleria: [
    'hotel_id','proveedor_id','hab_simples','hab_dobles','contratos_json','estado_registro','updated_at','observaciones'
  ],
  Acuerdos: [
    'acuerdo_id','proveedor_id','titulo','detalle','fecha_inicio','fecha_fin','monto_referencial','estado_registro','updated_at','observaciones'
  ],
  Programas: [
    'programa_id','proveedor_id','nombre_programa','inicio','fin','activo','estado_registro','updated_at','observaciones'
  ],
  Cambios_Log: [
    'cambio_id','fecha_hora','usuario','origen','entidad','entidad_id','accion','resumen','payload_json'
  ]
};

function doGet(e) {
  try {
    ensureSchema_();
    const action = getParam_(e, 'action', 'ping');

    switch (action) {
      case 'ping':
        return json_({ ok: true, ts: nowIso_(), version: APP_VERSION });
      case 'getCatalogos':
        return json_({ ok: true, catalogos: getCatalogos_() });
      case 'listBundle':
        return json_(listBundle_());
      case 'getProveedor':
        return json_(getProveedorBundle_(getParam_(e, 'proveedor_id', '')));
      default:
        return json_({ ok: false, error: 'Acción GET no soportada: ' + action });
    }
  } catch (err) {
    return json_({ ok: false, error: err.message, stack: err.stack });
  }
}

function doPost(e) {
  try {
    ensureSchema_();
    const payload = parseBody_(e);
    const action = payload.action || '';

    switch (action) {
      case 'upsertProveedorBundle':
        return json_(upsertProveedorBundle_(payload));
      case 'deleteProveedor':
        return json_(deleteProveedor_(payload));
      default:
        return json_({ ok: false, error: 'Acción POST no soportada: ' + action });
    }
  } catch (err) {
    return json_({ ok: false, error: err.message, stack: err.stack });
  }
}

function upsertProveedorBundle_(payload) {
  const actor = payload.actor || 'sistema';
  const proveedor = payload.proveedor || {};
  const contactos = Array.isArray(payload.contactos) ? payload.contactos : [];
  const hoteleria = payload.hoteleria || null;
  const acuerdos = Array.isArray(payload.acuerdos) ? payload.acuerdos : [];
  const programas = Array.isArray(payload.programas) ? payload.programas : [];

  const rutEmpresa = normalizeRut_(proveedor.rut_empresa || '');
  if (!rutEmpresa && !proveedor.proveedor_id) {
    throw new Error('Debes enviar proveedor_id o rut_empresa.');
  }

  const provSheet = getSheet_(SHEETS.proveedores);
  const provIndex = buildIndex_(provSheet, 'proveedor_id');
  const rutIndex = buildIndex_(provSheet, 'rut_empresa', true);
  const now = nowIso_();

  let proveedorId = proveedor.proveedor_id || '';
  let rowNum = proveedorId && provIndex[proveedorId] ? provIndex[proveedorId] : null;
  if (!rowNum && rutEmpresa && rutIndex[rutEmpresa]) rowNum = rutIndex[rutEmpresa];

  let current = rowNum ? rowToObject_(provSheet, rowNum) : null;
  if (!proveedorId) proveedorId = (current && current.proveedor_id) || createProveedorId_(rutEmpresa, proveedor.razon_social || proveedor.nombre_fantasia || 'proveedor');

  const mergedProv = {
    proveedor_id: proveedorId,
    rut_empresa: rutEmpresa || (current && current.rut_empresa) || '',
    razon_social: proveedor.razon_social || (current && current.razon_social) || '',
    nombre_fantasia: proveedor.nombre_fantasia || (current && current.nombre_fantasia) || '',
    localidad: proveedor.localidad || (current && current.localidad) || '',
    direccion: proveedor.direccion || (current && current.direccion) || '',
    correo_empresa: proveedor.correo_empresa || proveedor.correo || (current && current.correo_empresa) || '',
    fono_empresa: proveedor.fono_empresa || proveedor.fono || (current && current.fono_empresa) || '',
    giros_sii: normalizeTextArray_(proveedor.giros_sii || (current && current.giros_sii) || '').join('\n'),
    rubros_norm: normalizeTextArray_(proveedor.rubros_norm || (current && current.rubros_norm) || '').join('|'),
    actividad_principal: proveedor.actividad_principal || (current && current.actividad_principal) || '',
    descripcion_general: proveedor.descripcion_general || proveedor.descripcion || (current && current.descripcion_general) || '',
    plataformas_mineras: normalizeTextArray_(proveedor.plataformas_mineras || (current && current.plataformas_mineras) || '').join('|'),
    categoria_sii: proveedor.categoria_sii || (current && current.categoria_sii) || '',
    estado_facturacion: normalizeFacturacion_(proveedor.estado_facturacion || proveedor.facturar || (current && current.estado_facturacion) || ''),
    agrupacion_gremial: proveedor.agrupacion_gremial || proveedor.agrupacion || (current && current.agrupacion_gremial) || '',
    servicios_am: proveedor.servicios_am || proveedor.servicio_am || (current && current.servicios_am) || '',
    estado_registro: proveedor.estado_registro || (current && current.estado_registro) || 'Activo',
    fuente: proveedor.fuente || (current && current.fuente) || 'manual',
    created_at: (current && current.created_at) || now,
    updated_at: now,
    deleted_at: proveedor.deleted_at || (current && current.deleted_at) || '',
    observaciones: proveedor.observaciones || (current && current.observaciones) || ''
  };

  const values = objectToRow_(HEADERS.Proveedores, mergedProv);
  if (rowNum) provSheet.getRange(rowNum, 1, 1, values.length).setValues([values]);
  else rowNum = appendRow_(provSheet, values);

  replaceChildren_(SHEETS.contactos, 'proveedor_id', proveedorId, contactos.map(function (c, idx) {
    return {
      contacto_id: c.contacto_id || createContactoId_(proveedorId, c.rut_persona || c.nombre || String(idx + 1)),
      proveedor_id: proveedorId,
      rut_persona: normalizeRut_(c.rut_persona || c.rut || ''),
      nombre: c.nombre || '',
      cargo: c.cargo || '',
      correo: c.correo || '',
      fono: c.fono || '',
      principal: String(Boolean(c.principal || idx === 0)).toUpperCase(),
      estado_registro: c.estado_registro || 'Activo',
      created_at: c.created_at || now,
      updated_at: now,
      deleted_at: c.deleted_at || '',
      observaciones: c.observaciones || ''
    };
  }), HEADERS.Contactos);

  replaceChildren_(SHEETS.acuerdos, 'proveedor_id', proveedorId, acuerdos.map(function (a, idx) {
    return {
      acuerdo_id: a.acuerdo_id || ('acu_' + proveedorId + '_' + (idx + 1)),
      proveedor_id: proveedorId,
      titulo: a.titulo || '',
      detalle: a.detalle || '',
      fecha_inicio: a.fecha_inicio || '',
      fecha_fin: a.fecha_fin || '',
      monto_referencial: a.monto_referencial || '',
      estado_registro: a.estado_registro || 'Activo',
      updated_at: now,
      observaciones: a.observaciones || ''
    };
  }), HEADERS.Acuerdos);

  replaceChildren_(SHEETS.programas, 'proveedor_id', proveedorId, programas.map(function (p, idx) {
    return {
      programa_id: p.programa_id || ('prog_' + proveedorId + '_' + (idx + 1)),
      proveedor_id: proveedorId,
      nombre_programa: p.nombre_programa || p.nombre || '',
      inicio: p.inicio || '',
      fin: p.fin || '',
      activo: String(Boolean(p.activo !== false)).toUpperCase(),
      estado_registro: p.estado_registro || 'Activo',
      updated_at: now,
      observaciones: p.observaciones || ''
    };
  }), HEADERS.Programas);

  if (hoteleria) {
    replaceChildren_(SHEETS.hoteleria, 'proveedor_id', proveedorId, [{
      hotel_id: hoteleria.hotel_id || ('hot_' + proveedorId),
      proveedor_id: proveedorId,
      hab_simples: Number(hoteleria.hab_simples || hoteleria.simples || 0),
      hab_dobles: Number(hoteleria.hab_dobles || hoteleria.dobles || 0),
      contratos_json: safeStringify_(hoteleria.contratos || hoteleria.contratos_json || []),
      estado_registro: hoteleria.estado_registro || 'Activo',
      updated_at: now,
      observaciones: hoteleria.observaciones || ''
    }], HEADERS.Hoteleria);
  }

  logCambio_(actor, payload.actor === 'chatgpt' ? 'chatgpt' : 'app', 'proveedor', proveedorId, rowNum ? 'UPSERT' : 'CREATE', 'Alta/edición de bundle', payload);

  return { ok: true, proveedor_id: proveedorId, updated_at: now };
}

function deleteProveedor_(payload) {
  const proveedorId = payload.proveedor_id || '';
  const actor = payload.actor || 'sistema';
  if (!proveedorId) throw new Error('Falta proveedor_id.');

  const now = nowIso_();
  const provSheet = getSheet_(SHEETS.proveedores);
  const provIndex = buildIndex_(provSheet, 'proveedor_id');
  const rowNum = provIndex[proveedorId];
  if (!rowNum) throw new Error('Proveedor no encontrado: ' + proveedorId);

  const prov = rowToObject_(provSheet, rowNum);
  prov.estado_registro = 'Eliminado';
  prov.deleted_at = now;
  prov.updated_at = now;
  provSheet.getRange(rowNum, 1, 1, HEADERS.Proveedores.length).setValues([objectToRow_(HEADERS.Proveedores, prov)]);

  markChildrenDeleted_(SHEETS.contactos, proveedorId, now);
  markChildrenDeleted_(SHEETS.hoteleria, proveedorId, now);
  markChildrenDeleted_(SHEETS.acuerdos, proveedorId, now);
  markChildrenDeleted_(SHEETS.programas, proveedorId, now);

  logCambio_(actor, payload.actor === 'chatgpt' ? 'chatgpt' : 'app', 'proveedor', proveedorId, 'DELETE', 'Soft delete proveedor', payload);
  return { ok: true, proveedor_id: proveedorId, deleted_at: now };
}

function listBundle_() {
  const proveedores = listObjects_(SHEETS.proveedores).filter(function (r) { return r.estado_registro !== 'Eliminado'; });
  const contactos = listObjects_(SHEETS.contactos).filter(function (r) { return r.estado_registro !== 'Eliminado'; });
  const hoteleria = listObjects_(SHEETS.hoteleria).filter(function (r) { return r.estado_registro !== 'Eliminado'; });
  const acuerdos = listObjects_(SHEETS.acuerdos).filter(function (r) { return r.estado_registro !== 'Eliminado'; });
  const programas = listObjects_(SHEETS.programas).filter(function (r) { return r.estado_registro !== 'Eliminado'; });
  return { ok: true, proveedores: proveedores, contactos: contactos, hoteleria: hoteleria, acuerdos: acuerdos, programas: programas, catalogos: getCatalogos_() };
}

function getProveedorBundle_(proveedorId) {
  if (!proveedorId) throw new Error('Falta proveedor_id.');
  const proveedores = listObjects_(SHEETS.proveedores);
  const proveedor = proveedores.find(function (p) { return p.proveedor_id === proveedorId; });
  if (!proveedor) throw new Error('Proveedor no encontrado: ' + proveedorId);

  return {
    ok: true,
    proveedor: proveedor,
    contactos: listObjects_(SHEETS.contactos).filter(function (x) { return x.proveedor_id === proveedorId && x.estado_registro !== 'Eliminado'; }),
    hoteleria: listObjects_(SHEETS.hoteleria).find(function (x) { return x.proveedor_id === proveedorId && x.estado_registro !== 'Eliminado'; }) || null,
    acuerdos: listObjects_(SHEETS.acuerdos).filter(function (x) { return x.proveedor_id === proveedorId && x.estado_registro !== 'Eliminado'; }),
    programas: listObjects_(SHEETS.programas).filter(function (x) { return x.proveedor_id === proveedorId && x.estado_registro !== 'Eliminado'; })
  };
}

function replaceChildren_(sheetName, key, keyValue, rows, header) {
  const sheet = getSheet_(sheetName);
  const headers = header || HEADERS[sheetName];
  const all = sheet.getDataRange().getValues();
  if (all.length < 2) {
    if (rows.length) appendRows_(sheet, rows.map(function (r) { return objectToRow_(headers, r); }));
    return;
  }

  const headerRow = all[0];
  const idxKey = headerRow.indexOf(key);
  const keep = [];
  for (var i = 1; i < all.length; i++) {
    if (String(all[i][idxKey] || '') !== String(keyValue)) keep.push(all[i]);
  }

  sheet.clearContents();
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  if (keep.length) sheet.getRange(2, 1, keep.length, headers.length).setValues(keep);
  if (rows.length) appendRows_(sheet, rows.map(function (r) { return objectToRow_(headers, r); }));
}

function markChildrenDeleted_(sheetName, proveedorId, now) {
  const sheet = getSheet_(sheetName);
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return;
  const header = data[0];
  const idxProv = header.indexOf('proveedor_id');
  const idxEstado = header.indexOf('estado_registro');
  const idxUpdated = header.indexOf('updated_at');
  const idxDeleted = header.indexOf('deleted_at');

  for (var i = 1; i < data.length; i++) {
    if (String(data[i][idxProv] || '') === String(proveedorId)) {
      if (idxEstado >= 0) data[i][idxEstado] = 'Eliminado';
      if (idxUpdated >= 0) data[i][idxUpdated] = now;
      if (idxDeleted >= 0) data[i][idxDeleted] = now;
    }
  }
  sheet.getRange(1, 1, data.length, header.length).setValues(data);
}

function getCatalogos_() {
  const rows = listObjects_(SHEETS.catalogos);
  return rows.reduce(function (acc, row) {
    const key = row['Catálogo'] || row.catalogo || row.Catálogo;
    const val = row['Valor'] || row.valor || row.Valor;
    if (!key || !val) return acc;
    if (!acc[key]) acc[key] = [];
    acc[key].push(val);
    return acc;
  }, {});
}

function listObjects_(sheetName) {
  const sheet = getSheet_(sheetName);
  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];
  const header = values[0].map(String);
  return values.slice(1).filter(function (row) {
    return row.some(function (cell) { return String(cell || '').trim() !== ''; });
  }).map(function (row) {
    const obj = {};
    header.forEach(function (key, idx) { obj[key] = row[idx]; });
    return obj;
  });
}

function buildIndex_(sheet, key, normalizeRut) {
  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return {};
  const header = values[0].map(String);
  const idx = header.indexOf(key);
  if (idx < 0) throw new Error('No existe la columna ' + key + ' en ' + sheet.getName());
  const map = {};
  for (var i = 1; i < values.length; i++) {
    var val = values[i][idx];
    if (normalizeRut) val = normalizeRut_(val);
    if (String(val || '').trim()) map[val] = i + 1;
  }
  return map;
}

function rowToObject_(sheet, rowNum) {
  const header = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(String);
  const row = sheet.getRange(rowNum, 1, 1, header.length).getValues()[0];
  const obj = {};
  header.forEach(function (key, idx) { obj[key] = row[idx]; });
  return obj;
}

function objectToRow_(header, obj) {
  return header.map(function (key) { return obj[key] !== undefined ? obj[key] : ''; });
}

function appendRow_(sheet, values) {
  const rowNum = Math.max(sheet.getLastRow(), 1) + 1;
  sheet.getRange(rowNum, 1, 1, values.length).setValues([values]);
  return rowNum;
}

function appendRows_(sheet, rows) {
  if (!rows.length) return;
  const rowNum = Math.max(sheet.getLastRow(), 1) + 1;
  sheet.getRange(rowNum, 1, rows.length, rows[0].length).setValues(rows);
}

function ensureSchema_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  Object.keys(HEADERS).forEach(function (sheetName) {
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) sheet = ss.insertSheet(sheetName);
    var headers = HEADERS[sheetName];
    var firstRow = sheet.getRange(1, 1, 1, headers.length).getValues()[0];
    var empty = firstRow.every(function (x) { return String(x || '').trim() === ''; });
    if (empty) {
      sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
      styleHeader_(sheet, headers.length);
    }
  });
}

function styleHeader_(sheet, length) {
  sheet.getRange(1, 1, 1, length)
    .setBackground('#00778A')
    .setFontColor('#FFFFFF')
    .setFontWeight('bold')
    .setWrap(true);
  sheet.setFrozenRows(1);
}

function getSheet_(name) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  if (!sheet) throw new Error('No existe la hoja: ' + name);
  return sheet;
}

function parseBody_(e) {
  const raw = e && e.postData && e.postData.contents ? e.postData.contents : '{}';
  return JSON.parse(raw);
}

function getParam_(e, key, fallback) {
  return (e && e.parameter && e.parameter[key] !== undefined) ? e.parameter[key] : fallback;
}

function json_(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function normalizeRut_(value) {
  return String(value || '').replace(/[^0-9kK]/g, '').toUpperCase();
}

function normalizeTextArray_(value) {
  if (Array.isArray(value)) return value.map(String).map(function (s) { return s.trim(); }).filter(Boolean);
  return String(value || '').split(/[\n|;,]+/).map(function (s) { return s.trim(); }).filter(Boolean);
}

function normalizeFacturacion_(value) {
  const v = String(value || '').trim().toLowerCase();
  if (!v) return 'Sin información';
  if (v === 'autorizado') return 'Autorizado';
  if (v === 'boleta honorario' || v === 'boleta honorarios' || v === 'honorarios') return 'Boleta honorario';
  if (v === 'no' || v === 'no autorizado' || v === 'sin autorización' || v === 'sin autorizacion') return 'NO';
  if (v === 'en trámite' || v === 'en tramite' || v === 'pendiente') return 'Sin información';
  return value;
}

function createProveedorId_(rutEmpresa, fallbackName) {
  if (rutEmpresa) return 'prov_re_' + rutEmpresa;
  const slug = String(fallbackName || 'proveedor').toLowerCase().replace(/[^a-z0-9]+/g, '_').replace(/^_|_$/g, '').slice(0, 24);
  return 'prov_nm_' + slug;
}

function createContactoId_(proveedorId, seed) {
  const clean = normalizeRut_(seed);
  if (clean) return 'cont_rp_' + clean;
  const slug = String(seed || 'contacto').toLowerCase().replace(/[^a-z0-9]+/g, '_').replace(/^_|_$/g, '').slice(0, 20);
  return 'cont_' + proveedorId + '_' + slug;
}

function safeStringify_(value) {
  return typeof value === 'string' ? value : JSON.stringify(value || []);
}

function nowIso_() {
  return Utilities.formatDate(new Date(), TZ, "yyyy-MM-dd'T'HH:mm:ss");
}

function logCambio_(usuario, origen, entidad, entidadId, accion, resumen, payload) {
  const sheet = getSheet_(SHEETS.cambios);
  appendRow_(sheet, [
    'log_' + new Date().getTime(),
    nowIso_(),
    usuario || 'sistema',
    origen || 'app',
    entidad || '',
    entidadId || '',
    accion || '',
    resumen || '',
    safeStringify_(payload)
  ]);
}
